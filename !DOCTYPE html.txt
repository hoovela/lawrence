  <script src="https://unpkg.com/jspsych@7.3.0"></script>
  <script src="https://unpkg.com/@jspsych/plugin-html-keyboard-response"></script>
  <link href="https://unpkg.com/jspsych@7.3.0/css/jspsych.css" rel="stylesheet" />
</head>
<body></body>
<script>

const lawrence = ["Lawrence","City council","Essex Street","South Lawrence","The Loop","Multi-family home","Gateway"];
const andover = ["North Andover","Town meeting","Route 114","Turnpike St","Merrimack College","Single-family home","Rolf’s"];
const positive = ["Peaceful","Safe","Clean","Friendly","Quiet","Thriving","Charming"];
const negative = ["Dangerous","Chaotic","Unsafe","Dirty","Crowded","Noisy","Run-down"];

function makeTrials(leftItems, rightItems, leftKey='e', rightKey='i', n=20){
  let trials = [];
  for(let i=0;i<n;i++){
    let isLeft = Math.random() < 0.5;
    let word = isLeft 
      ? leftItems[Math.floor(Math.random()*leftItems.length)]
      : rightItems[Math.floor(Math.random()*rightItems.length)];

    trials.push({
      type: jsPsychHtmlKeyboardResponse,
      stimulus: `<h1>${word}</h1>
                 <p>${leftKey.toUpperCase()} = LEFT | ${rightKey.toUpperCase()} = RIGHT</p>`,
      choices: [leftKey, rightKey],
      data: {correct: isLeft ? leftKey : rightKey},
      on_finish: function(data){
        data.correct_response = data.response == data.correct;
      }
    });
  }
  return trials;
}

let timeline = [];

// Instructions
timeline.push({
  type: jsPsychHtmlKeyboardResponse,
  stimulus: "<p>Press E for LEFT category, I for RIGHT category.<br>Respond as quickly as possible.</p><p>Press any key to begin.</p>"
});

// Blocks
timeline.push(...makeTrials(lawrence, andover));
timeline.push(...makeTrials(positive, negative));
timeline.push(...makeTrials(lawrence.concat(positive), andover.concat(negative)));
timeline.push(...makeTrials(lawrence.concat(positive), andover.concat(negative), 'e','i',40));
timeline.push(...makeTrials(andover, lawrence));
timeline.push(...makeTrials(andover.concat(positive), lawrence.concat(negative)));
timeline.push(...makeTrials(andover.concat(positive), lawrence.concat(negative),'e','i',40));

// Run
const jsPsych = initJsPsych({
  on_finish: function(){
    jsPsych.data.displayData();
  }
});

jsPsych.run(timeline);

</script>
</html>
